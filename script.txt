# Policy routing (once)
```bash
sudo sysctl -w net.ipv4.ip_forward=1
sudo ip rule add fwmark 1 lookup 100
sudo ip route add local 0.0.0.0/0 dev lo table 100
sudo nft add table inet tproxy2
```
# Clean just your tproxy2 table so we start fresh
```bash
sudo nft flush table inet tproxy2
```

# Ensure chains exist at the correct hook/priority (priority "mangle" == -150)
```bash
sudo nft add chain inet tproxy2 divert  '{ type filter hook prerouting priority mangle; }'
sudo nft add chain inet tproxy2 preroute '{ type filter hook prerouting priority mangle; }'
```
# 1) Divert rule: attach to already-matched transparent sockets and set the fwmark
```bash
sudo nft add rule inet tproxy2 divert \
  meta l4proto udp socket transparent 1 meta mark set 0x1 accept
```
# 2) TPROXY rule: intercept UDP/24680 from Wi-Fi clients and hand to your proxy
```bash
sudo nft add rule inet tproxy2 preroute \
  iifname "wlan0" udp dport 24680 \
  tproxy to :19001 meta mark set 0x1 accept
```

sudo nft create chain inet tproxy2 preroute \
    '{ type filter hook prerouting priority mangle; policy accept; }' 2>/dev/null || true

# Flush to ensure exact rule order (optional but makes it match your working sample)
